Java实现随机字符串验证码

随机生成由大写字母或小写字母或数字或汉字组成的字符串

随机验证码的优缺点

  • 优点: 生成方便，识别难易度可调节
  • 缺点: 识别度不好把握

随机字符串验证码的实现方式

  • Java实现

    • 过程

      1. 用户每次访问页面时随机生成一张图片，图片内容通常是数字，汉字或字母的随机组合
      2. 访问者识别图片的内容并填写表单提交
      3. 服务器端将接收到的访问者提交的验证码与Session范围内保存的验证码进行比较，进行合法性判断

    • 原理

      1. 创建画板：new BufferedImage()
      2. 创建画笔：画板.getGraphics()
      3. 随机生成内容：不同颜色的字符
      4. 绘制内容：画笔drawString()
      5. 存为图片发送：ImageIO.wirte(画板，图片类型，输出流)

    • 代码实现
      创建randomCode.class

      public class randomCode {

    private Random random = new Random(); // 随机数对象
    private int width; // 图片宽度
    private int height; // 图片高度
    private int fontsize; // 字体大小
    private String str = "0123456789qwertyuioplkjhgfdsazxcvbnm"; // 字符集
    private String compareStr; // 验证比较字符

    public String getCompareStr() {
        return compareStr;
    }

    public void setCompareStr(String compareStr) {
        this.compareStr = compareStr;
    }

    public int getWidth() {
        return width;
    }

    public void setWidth(int width) {
        this.width = width;
    }

    public int getHeight() {
        return height;
    }

    public void setHeight(int height) {
        this.height = height;
    }

    public int getFontsize() {
        return fontsize;
    }

    public void setFontsize(int fontsize) {
        this.fontsize = fontsize;
    }

    // 生成最短长度为4的字符串
    private String randCode(int len) {
        if (len < 4) {
            len = 4;
        }
        width = 5 + fontsize * len;
        String code = "";
        for (int i = 0; i < len; i++) {
            code += str.charAt(random.nextInt(str.length()));
        }
        return code;
    }

    // 返回随机颜色
    private Color randomColor() {
        int r = random.nextInt(256);
        int g = random.nextInt(256);
        int b = random.nextInt(256);
        return new Color(r, g, b);
    }

    // 画板元素具体值
    public void createImgElement(int width, int height, int fontsize) {
        setWidth(width);
        setHeight(height);
        setFontsize(fontsize);

    }

    public BufferedImage randomCodeImg() {
        // 1、创建画板
        BufferedImage img = new BufferedImage(width, height, BufferedImage.TYPE_INT_RGB);
        // 2、创建画笔
        Graphics2D pen = (Graphics2D) img.getGraphics();
        // 3、生成随机内容
        String codeStr = randCode(4);
        // 将生成内容赋值给比较验证字符
        setCompareStr(codeStr);
        // 4、绘制内容
        // 4.1设置绘制区域
        pen.fillRect(0, 0, width, height);
        // 4.2设置字体
        pen.setFont(new Font("微软雅黑", Font.BOLD, fontsize));
        // 4.3按顺序逐个绘制字符
        for (int i = 0; i < codeStr.length(); i++) {
            pen.setColor(randomColor());
            pen.drawString(codeStr.charAt(i) + "", 5 + i * fontsize, (fontsize + height) / 2);
        }
        // 4.4绘制噪音线
        for (int i = 0; i < 3; i++) {
            pen.setBackground(randomColor());
            pen.setStroke(new BasicStroke(2));
            pen.drawLine(random.nextInt(width / 2), random.nextInt(height), random.nextInt(width),
                    random.nextInt(height));
        }
        // 5返回图片
        return img;
    }

    • 代码实现

      //randomcode.js  
/*使用方式：html或jsp页面中使用canvas标签，利用js获取canvas标签ID为 var canvas，调用  drawcode（canvas）方法；
实现例子：
 <script type="text/javascript">
  var code; //接受返回的随机字符串值
  function changeCode(){
    var cvs = document.getElementById("cvs");
    code = drawcode(cvs);
  }
 </script>
*/
//高度、宽度、字符大小、字符集
var w = 80;
var h = 24;
var fontsize = h-6;
var str = "0123456789abcdefABCDEF";

//随机生成最大值不超过max的整数
function randInt(max){
  return Math.floor(Math.random()*100000%max);
}

//生成随机长度的字符串验证码
function randCode(len){
  if(len<4){
      len = 4;
  }
  var code = "";
  for(var i=0; i<len; i++){
      code += str.charAt(randInt(str.length));
  }
  return code;
}

//生成随机颜色
function randColor(){
  var r = randInt(256);
  var g = randInt(256);
  var b = randInt(256);
  return "rgb("+r+","+g+","+b+")";
}

//绘制图片 return code  code为传递变量，在jsp或html页面中调用，用来与用户输入的验证码制作比较
function drawcode(canvas){
  var code = randCode(4);
  w = 5 + fontsize*code.length;

  if(canvas != null && canvas.getContext && canvas.getContext("2d")){
      //设置显示区域大小
      canvas.style.width = w;
      //设置画板高宽
      canvas.setAttribute("width",w);
      canvas.setAttribute("height",h);
      //得到画笔
      var pen = canvas.getContext("2d");
      //绘制背景
      pen.fillStyle = "rgb(255,255,255)";
      pen.fillRect(0,0,w,h);
      //设置水平线位置
      pen.textBaseline = "top";    //middle,bottom
      //绘制内容
      for(var i=0; i<code.length; i++){
          pen.fillStyle = randColor();
          pen.font = "bold "+(fontsize+randInt(3))+"px 微软雅黑";
          pen.fillText(code.charAt(i), 5+fontsize*i, randInt(5));
      }
      //绘制噪音线
      for(var i=0; i<2; i++){
          pen.moveTo(randInt(w)/2, randInt(h));    //设置起点
          pen.lineTo(randInt(w), randInt(h));        //设置终点
          pen.strokeStyle = randColor();
          pen.lineWidth = 2;        //线条粗细
          pen.stroke();
      }
      return code;
  }
}
